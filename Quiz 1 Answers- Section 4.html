list_datasets = ["data_quiz/wdi_co2_2005.csv","data_quiz/wdi_co2_2010.csv","data_quiz/wdi_co2_2015.csv"]

index = 0
for dataset_name in list_datasets:
    dataset = pd.read_csv(dataset_name)
    co2_gasfuel_kt    = dataset["co2_gasfuel_kt"]
    co2_liquidfuel_kt = dataset["co2_liquidfuel_kt"] 
    co2_fuel_kt       = co2_gasfuel_kt + co2_liquidfuel_kt
    plt.hist(co2_fuel_kt)
    plt.title("Distribution for :" + dataset_name)
    plt.ylabel("Frequency")
    plt.xlabel("CO2 emissions from fuel")
    plt.show()

    index = index + 1
